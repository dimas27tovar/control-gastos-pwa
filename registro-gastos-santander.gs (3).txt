/**
 * REGISTRO AUTOMÁTICO DE GASTOS — Santander → Google Sheets
 * ------------------------------------------------------------
 * Busca correos de avisos de transacción/transferencia de Santander,
 * extrae monto, establecimiento y fecha, y los guarda como fila nueva
 * en una hoja de cálculo. Marca cada correo como procesado para no
 * duplicarlo.
 *
 * INSTALACIÓN (una sola vez):
 * 1. Ve a https://script.google.com → "Nuevo proyecto"
 * 2. Borra el contenido de ejemplo y pega todo este archivo
 * 3. Cambia SHEET_ID (línea ~20) por el ID de tu Google Sheet
 *    (lo sacas de la URL: docs.google.com/spreadsheets/d/ESTE_ES_EL_ID/edit)
 *    Si no tienes una hoja aún, crea una en sheets.new y copia su ID.
 * 4. Guarda el proyecto (ícono de disco, ponle un nombre)
 * 5. En el menú de funciones (arriba), selecciona "configurarTrigger"
 *    y presiona ▶ Ejecutar. Te va a pedir autorizar permisos de
 *    Gmail y Sheets — acepta (verás una advertencia de "app no
 *    verificada", es normal porque es tu propio script; dale
 *    "Avanzado" → "Ir a [nombre] (no seguro)")
 * 6. Listo. El script correrá solo cada 15 minutos a partir de ahora.
 *
 * Para probarlo manualmente sin esperar: selecciona "procesarCorreosBanco"
 * en el menú de funciones y presiona ▶ Ejecutar.
 */

// ==== CONFIGURACIÓN ====
const SHEET_ID = "1Zyfk5cMcE7G3PPaQZ9UO3JL0oCQGfGEnES891vh9H9k";
const SHEET_NAME = "Gastos";
const LABEL_PROCESADO = "GastosApp-Procesado";
const REMITENTE = "santander"; // ajusta si el correo viene de otro dominio

// ==== FUNCIÓN PRINCIPAL ====
function procesarCorreosBanco() {
  const label = getOrCreateLabel_(LABEL_PROCESADO);
  const query = `from:(${REMITENTE}) -label:${LABEL_PROCESADO} newer_than:14d`;
  const threads = GmailApp.search(query, 0, 50);

  if (threads.length === 0) return;

  const sheet = getSheet_();

  threads.forEach((thread) => {
    const messages = thread.getMessages();
    messages.forEach((message) => {
      const asunto = message.getSubject();
      const cuerpo = message.getPlainBody();
      const datos = extraerDatos_(asunto, cuerpo);

      if (datos) {
        sheet.appendRow([
          datos.fecha,
          datos.hora,
          datos.tipo,
          datos.establecimiento,
          datos.monto,
          datos.moneda,
          asunto,
        ]);
      }
    });
    thread.addLabel(label);
  });
}

// ==== EXTRACCIÓN DE DATOS DEL CORREO ====
function extraerDatos_(asunto, cuerpo) {
  // Monto: soporta "USD 2,41", "$ 1.234,56", "MXN 450.00", etc.
  const matchImporte = cuerpo.match(/Importe\s*\n?\s*([A-Z$]{0,3})\s*([\d.,]+)/i);
  if (!matchImporte) return null; // si no encuentra monto, no es un correo de transacción reconocible

  const moneda = (matchImporte[1] || "").trim() || "N/D";
  const monto = parseMonto_(matchImporte[2]);

  // Establecimiento o beneficiario (compra vs transferencia)
  // Nota: Santander escribe "Establecimento" (sin la "i"), por eso el patrón
  // acepta ambas variantes: Establecim(i)ento
  const matchEstablecimiento = cuerpo.match(
    /(?:Establecim(?:i)?ento|Comercio|Beneficiario|Destinatario)\s*\n?\s*([^\n]{1,40})/i
  );
  let establecimiento = matchEstablecimiento
    ? matchEstablecimiento[1].trim()
    : "No identificado";

  // Validación: si lo capturado parece prosa (frase larga con conectores
  // como "de", "su", "para") en vez de un nombre de comercio, se descarta
  if (/\b(de|su|para|el|la|los|las|por)\b/i.test(establecimiento) && establecimiento.split(" ").length > 3) {
    establecimiento = "No identificado";
  }

  // Fecha y hora
  const matchFecha = cuerpo.match(
    /Fecha y hora\s*\n?\s*(\d{2}\/\d{2}\/\d{4})\s*\n?\s*(\d{2}:\d{2}:\d{2})/i
  );
  const fecha = matchFecha ? matchFecha[1] : Utilities.formatDate(new Date(), "GMT-3", "dd/MM/yyyy");
  const hora = matchFecha ? matchFecha[2] : "";

  // Tipo de movimiento, según el asunto (sin depender de acentos)
  let tipo = "Movimiento";
  const asuntoSinAcentos = quitarAcentos_(asunto.toLowerCase());
  if (asuntoSinAcentos.includes("transferencia")) tipo = "Transferencia";
  else if (asuntoSinAcentos.includes("debito")) tipo = "Compra d\u00e9bito";
  else if (asuntoSinAcentos.includes("credito")) tipo = "Compra cr\u00e9dito";
  else if (asuntoSinAcentos.includes("transaccion")) tipo = "Compra";

  return { fecha, hora, tipo, establecimiento, monto, moneda };
}

// Quita acentos para comparar texto sin depender de codificación
function quitarAcentos_(texto) {
  return texto.normalize("NFD").replace(/[\u0300-\u036f]/g, "");
}

// Convierte "2,41" o "1.234,56" a número 2.41 / 1234.56
function parseMonto_(texto) {
  let limpio = texto.trim();
  const tieneComaYPunto = limpio.includes(",") && limpio.includes(".");
  if (tieneComaYPunto) {
    // el que aparece último es el decimal
    if (limpio.lastIndexOf(",") > limpio.lastIndexOf(".")) {
      limpio = limpio.replace(/\./g, "").replace(",", ".");
    } else {
      limpio = limpio.replace(/,/g, "");
    }
  } else if (limpio.includes(",")) {
    limpio = limpio.replace(",", ".");
  }
  return parseFloat(limpio) || 0;
}

// ==== UTILIDADES ====
function getSheet_() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow([
      "Fecha",
      "Hora",
      "Tipo",
      "Establecimiento",
      "Monto",
      "Moneda",
      "Asunto original",
    ]);
  }
  return sheet;
}

function getOrCreateLabel_(nombre) {
  return GmailApp.getUserLabelByName(nombre) || GmailApp.createLabel(nombre);
}

// ==== ENDPOINT WEB APP (para que la app del celular lea los datos) ====
function doGet(e) {
  const sheet = getSheet_();
  const datos = sheet.getDataRange().getValues();
  const encabezados = datos[0];
  const filas = datos.slice(1);

  const gastos = filas
    .filter((fila) => fila[0]) // ignora filas vacías
    .map((fila, index) => {
      const obj = { id: index + 2 }; // +2 porque la fila 1 son encabezados
      encabezados.forEach((encabezado, i) => {
        const clave = normalizarClave_(encabezado);
        obj[clave] = formatearValor_(clave, fila[i]);
      });
      return obj;
    })
    .reverse(); // más reciente primero

  return ContentService.createTextOutput(JSON.stringify(gastos)).setMimeType(
    ContentService.MimeType.JSON
  );
}

// Convierte objetos Date (que Sheets genera para columnas de fecha/hora)
// a texto legible, en vez de mandarlos como ISO crudo
function formatearValor_(clave, valor) {
  if (valor instanceof Date) {
    if (clave === "fecha") return Utilities.formatDate(valor, "GMT-3", "dd/MM/yyyy");
    if (clave === "hora") return Utilities.formatDate(valor, "GMT-3", "HH:mm:ss");
    return Utilities.formatDate(valor, "GMT-3", "dd/MM/yyyy HH:mm:ss");
  }
  return valor;
}

function normalizarClave_(texto) {
  return quitarAcentos_(String(texto).toLowerCase()).replace(/\s+/g, "_");
}

// ==== CONFIGURAR EJECUCIÓN AUTOMÁTICA ====
// Ejecuta esta función UNA sola vez para activar el trigger cada 15 min.
function configurarTrigger() {
  // evita duplicar triggers si ya existe uno
  ScriptApp.getProjectTriggers().forEach((t) => {
    if (t.getHandlerFunction() === "procesarCorreosBanco") {
      ScriptApp.deleteTrigger(t);
    }
  });
  ScriptApp.newTrigger("procesarCorreosBanco")
    .timeBased()
    .everyMinutes(15)
    .create();

  // corre una vez de inmediato para probar
  procesarCorreosBanco();
}
